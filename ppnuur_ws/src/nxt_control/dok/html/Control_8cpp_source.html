188 void Control::updatePosition(const nxt_control::SensorData& msg) {
189  _distance = msg.distance;
190  double wheel_L = 2.0 * M_PI * _wheel_radius_l * msg.tickCountA
191  / _enc_per_turn_left;
192  double wheel_R = (2.0) * M_PI * _wheel_radius_r * msg.tickCountB
193  / _enc_per_turn_right;
194  double dtheta = (wheel_R - wheel_L) / _axis_length * _turning_adaptation;
195  if (fabs(dtheta) < 0.005) {
196  double wheel = (wheel_L + wheel_R) / 2.0;
197  _pose.x += wheel * cos(_pose.th);
198  _pose.y += wheel * sin(_pose.th);
199  } else {
200  double radius = wheel_L / dtheta;
201  double cx = _pose.x - (radius + _axis_length / 2.0) * sin(_pose.th);
202  double cy = _pose.y + (radius + _axis_length / 2.0) * cos(_pose.th);
203  double newTheta = fmod(_pose.th + dtheta + 3 * M_PI, 2 * M_PI) - M_PI;
204  _pose.x = cx + (radius + _axis_length / 2.0) * sin(newTheta);
205  _pose.y = cy - (radius + _axis_length / 2.0) * cos(newTheta);
206  _pose.th = newTheta;
207  }
